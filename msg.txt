KEYS:
  win + d      launcher/window selector
  win + enter  new terminal
TERM:
  s r       # go to rhea w/X forwarding
 WALK:
  cd ..     # go up
  cd -      # jump back
  cd ~      # go home
  cd /      # go to root
  z pattern # go to dir matching 'pattern'
 EDIT:
  cltr + a  # go to start
  cltr + e  # go to end
  ctrl + w  # delete word back
  cltr + y  # paste deleted word
  ctrl + _  # undo
